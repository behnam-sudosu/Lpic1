# find  

	find ===>> locate files on system based on name, type, date, ...
	find	-user
			-size
			-type
			-group
			-delete
			
	find direction NAME or FILE
	find / -name "file1"
	find / -iname "file1" (i = uppercase and lowercase)
	find / -user behnam (find file or directory the owner is behnam)
	find / -group behnam (find file or directory the owner group is behnam)
	find / -size +10M -size -20M
	find / -delete behnam (delete file)
	find / -type f behnam (f = file)
	find / -type d behnam (d = directory)
	find / -type l behnam (l = link)
	find / -type perm 777 (find file that have permision 777)
	find / -perm -777 (find file that have permision 777)
	find / -exec ===>> you can add cammand as a script

	accesstime ===>> atime ===>> a
	modifytime ===>> mtime ===>> c
	changetime ===>> ctime ===>> m

	access
	find / -atime 6 ===>> 6 days ago and before that
	find / -atime -6 ===>> now until 6 days ago

	change (change owner)
	find / -ctime 6 ===>> 6 days ago and before that
	find / -ctime -6 ===>> now until 6 days ago

	modify
	find / -mtime 6 ===>> 6 days ago and before that
	find / -mtime -6 ===>> now until 6 days ago

	find / -amin -60 ===>> show minutes
	find / -cmin -60 ===>> show minutes

	find /var/log -type f -mtime +30 -delete ===>> delete file before 30 days
