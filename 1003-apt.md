# debian

```bash

wget LINK URL

wget -c FILE-NAME

dpkg -i PACKAGE NAME

dpkg -r vim

dpkg --contents vim

dpkg -s vim

# remove package
dpkg -P vim
dpkg --purge vim

dpkg -S /usr/share/doc/vim
```

### apt

```bash
# install package
apt

# change country
vim /etc/apt/sources.list.d
	deb htt://SERVER_IP

# update list of repository
apt update

# upgrade package
apt upgrade

# install package
apt install openssh-service

# show firefox in repository
apt-cache search firefox
apt-cache search netcat

# show package must upgrade
apt list --upgradable

# show library and where is download package
apt-cache showpkg nginx

# show dependency
apt-cache depends nginx
apt-cache depends telnet

# remove dependency
apt autoremove

# remove package
apt remove PACKAGE_NAME

# uninstall program
apt purge PACKAGE NAME

# clean cash
apt clean

# remove .deb old
apt autoclean

# upgrade your os
do-release-upgrade

# directory apt upgrade
/var/cache/apt/archives

# directort apt update
/var/lib/apt/lists
```

# redhat

### rpm

```bash
# download
wget LINK OR URL

# for install 
rpm -i PACKAGE NAME

# v = verbos
rpm -iv PACKAGE NAME

# h = #####
rpm -ih PACKAGE NAME

# query
rpm -q

# where copy file in directory
rpm -ql

# which package installed in this directory
rpm -qf /usr/share/share/locale/vim

# show more information about package
rpm -qip

# where is config file
rpm -qc

# all list package installed in your os
rpm -qa

# library needs
rpm -qR vim

# erase package
rpm -e PACKAGE_NAME

# if installed just upgrade and if dosn't installed just install
rpm -U

# is upgrade if installed (won't install)
rpm -F

# erase
rpm -e

# package is not chage, it's kind of sighne
rpm -K vim

# if download package show is download complete
rpm -k vim-4.0.0-6.fck

# verify package everything is correct
rpm -V vim
```

### yum

```bash
# chnage country
/etc/yum.repos.d
/etc/yum.conf

# install package
yum

# install package too
dnf

# dosn't upgrade when you install package yum first go upgrade all package

	keepcache=1

# upgrade
yum update

# upgrade
yum upgrade

# upgrade package start with z
yum upgrade 'z*'

# install vim package
yum install vim

# more information
yum info vim

# reinstall package if have problem
yum reinstall vim

# what package must install

# remove

# search

# show dpendency
yum deplist vim

# show list of install in your os
```
